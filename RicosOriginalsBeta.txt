  <script>
    let categories = ['Abenteuer', 'Fantasy'];
    let data = JSON.parse(localStorage.getItem('ricos_content')) || [];
    let isLoggedIn = false;

    function toggleLogin() {
      document.getElementById('login-panel').classList.toggle('hidden');
    }

    function checkLogin() {
      const pw = document.getElementById('admin-password').value;
      if (pw === 'admin123') {
        isLoggedIn = true;
        document.getElementById('login-panel').classList.add('hidden');
        document.getElementById('admin-panel').classList.remove('hidden');
        fillCategorySelect();
      } else {
        alert('Falsches Passwort!');
      }
    }

    function logout() {
      isLoggedIn = false;
      document.getElementById('admin-panel').classList.add('hidden');
    }

    function addCategory() {
      const cat = document.getElementById('new-category').value;
      if (cat && !categories.includes(cat)) {
        categories.push(cat);
        fillCategorySelect();
        alert('Kategorie hinzugefügt!');
      }
    }

    function fillCategorySelect() {
      const select = document.getElementById('category-select');
      select.innerHTML = '';
      categories.forEach(c => {
        const opt = document.createElement('option');
        opt.value = c;
        opt.textContent = c;
        select.appendChild(opt);
      });
    }

    function saveContent() {
      const type = document.getElementById('content-type').value;
      const title = document.getElementById('title').value;
      const category = document.getElementById('category-select').value;
      const thumb = document.getElementById('thumb').files[0];
      const video = document.getElementById('video').files[0];
      if (!title || !thumb || !video) return alert('Bitte alle Felder ausfüllen!');

      const readerThumb = new FileReader();
      const readerVideo = new FileReader();
      readerThumb.onload = () => {
        readerVideo.onload = () => {
          data.push({
            type, title, category,
            thumb: readerThumb.result,
            video: readerVideo.result
          });
          localStorage.setItem('ricos_content', JSON.stringify(data));
          alert('Gespeichert!');
          showCategory(type);
        };
        readerVideo.readAsDataURL(video);
      };
      readerThumb.readAsDataURL(thumb);
    }

    function showCategory(type) {
      document.getElementById('home').classList.add('hidden');
      const section = document.getElementById('category-section');
      section.innerHTML = `<h2>${type === 'film' ? 'Filme' : 'Serien'}</h2>`;
      section.classList.remove('hidden');

      const items = data.filter(item => item.type === type);
      items.forEach(item => {
        const div = document.createElement('div');
        div.className = 'card';
        div.innerHTML = `
          <img src="${item.thumb}" alt="${item.title}">
          <h3>${item.title}</h3>
          <button onclick="playVideo('${item.video}')">Abspielen</button>
        `;
        section.appendChild(div);
      });
    }

    function showHome() {
      document.getElementById('category-section').classList.add('hidden');
      document.getElementById('home').classList.remove('hidden');
    }

    function playVideo(videoSrc) {
      const win = window.open();
      win.document.write(`<video src="${videoSrc}" controls autoplay style="width:100%;height:100%"></video>`);
    }
